# 多人聊天系统数据库设计

## 系统概述

本系统支持一对一聊天、群组聊天、消息撤回、消息已读、未读消息数等功能，采用关系型数据库存储核心数据。

## 核心表结构设计

### 用户模块

```sql
-- 用户基础信息表
CREATE TABLE t_chat_user (
    id BIGINT NOT NULL AUTO_INCREMENT COMMENT '用户ID',
    username VARCHAR(50) NOT NULL COMMENT '用户名',
    password VARCHAR(255) NOT NULL COMMENT '密码',
    nickname VARCHAR(100) NOT NULL COMMENT '昵称',
    avatar_url VARCHAR(500) DEFAULT NULL COMMENT '头像URL',
    email VARCHAR(100) DEFAULT NULL COMMENT '邮箱',
    phone VARCHAR(20) DEFAULT NULL COMMENT '手机号',
    gender TINYINT DEFAULT 0 COMMENT '性别：0-未知，1-男，2-女',
    birthday DATE DEFAULT NULL COMMENT '生日',
    bio VARCHAR(255) DEFAULT NULL COMMENT '个人简介',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-离线，1-在线，2-隐身',
    last_active_at DATETIME DEFAULT NULL COMMENT '最后活跃时间',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    deleted_at DATETIME DEFAULT NULL COMMENT '删除时间',
    PRIMARY KEY (id),
    UNIQUE KEY uk_username (username),
    KEY idx_status (status),
    KEY idx_last_active (last_active_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 用户会话关系表（维护用户之间的好友关系）
CREATE TABLE t_chat_friend (
    id BIGINT NOT NULL AUTO_INCREMENT COMMENT '关系ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    friend_id BIGINT NOT NULL COMMENT '好友用户ID',
    friend_nickname VARCHAR(100) DEFAULT NULL COMMENT '好友备注名',
    friend_group VARCHAR(50) DEFAULT NULL COMMENT '好友分组',
    is_blacklist TINYINT NOT NULL DEFAULT 0 COMMENT '是否拉黑：0-否，1-是',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '添加时间',
    PRIMARY KEY (id),
    UNIQUE KEY uk_user_friend (user_id, friend_id),
    KEY idx_friend_id (friend_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='好友关系表';
```

### 会话模块

```sql
-- 会话表（单聊和群聊共用的会话记录）
CREATE TABLE t_chat_conversation (
    id BIGINT NOT NULL AUTO_INCREMENT COMMENT '会话ID',
    conversation_key VARCHAR(100) NOT NULL COMMENT '会话唯一标识',
    conversation_type TINYINT NOT NULL COMMENT '类型：1-单聊，2-群聊',
    owner_id BIGINT NOT NULL COMMENT '拥有者用户ID（用于快速查询用户的会话列表）',
    target_id BIGINT NOT NULL COMMENT '目标ID（单聊为对方用户ID，群聊为群ID）',
    chat_mode TINYINT NOT NULL DEFAULT 1 COMMENT '会话模式：1-文字，2-语音，3-视频',
    is_pinned TINYINT NOT NULL DEFAULT 0 COMMENT '是否置顶：0-否，1-是',
    is_muted TINYINT NOT NULL DEFAULT 0 COMMENT '是否静音：0-否，1-是',
    is_deleted TINYINT NOT NULL DEFAULT 0 COMMENT '是否删除：0-否，1-是',
    last_message_id BIGINT DEFAULT NULL COMMENT '最后一条消息ID',
    last_message_content VARCHAR(500) DEFAULT NULL COMMENT '最后一条消息摘要',
    last_message_time DATETIME DEFAULT NULL COMMENT '最后消息时间',
    unread_count INT NOT NULL DEFAULT 0 COMMENT '未读消息数',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (id),
    UNIQUE KEY uk_conversation_key (conversation_key),
    KEY idx_owner_conversations (owner_id, is_deleted, last_message_time),
    KEY idx_target (target_id, conversation_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='会话表';

-- 单聊会话Key生成规则：MIN(user_id_1, user_id_2) + "_" + MAX(user_id_1, user_id_2)
-- 示例：1和5的会话Key为 "1_5"

-- 群聊会话Key生成规则："group_" + group_id
-- 示例：群ID为100的会话Key为 "group_100"
```

-- 消息表
CREATE TABLE t_chat_message (
    id BIGINT NOT NULL AUTO_INCREMENT COMMENT '消息ID',
    message_key VARCHAR(100) NOT NULL COMMENT '消息唯一标识（客户端生成）',
    conversation_id BIGINT NOT NULL COMMENT '会话ID',
    conversation_type TINYINT NOT NULL COMMENT '会话类型：1-单聊，2-群聊',
    sender_id BIGINT NOT NULL COMMENT '发送者用户ID',
    sender_name VARCHAR(100) NOT NULL COMMENT '发送者昵称',
    receiver_id BIGINT NOT NULL COMMENT '接收者ID（用户ID或群ID）',
    message_type TINYINT NOT NULL COMMENT '消息类型：1-文本，2-图片，3-语音，4-视频，5-文件，6-位置，7-卡券，8-撤回，9-引用，10-表情',
    content TEXT DEFAULT NULL COMMENT '消息内容',
    content_preview VARCHAR(200) DEFAULT NULL COMMENT '消息预览（用于列表展示）',
    extra_data JSON DEFAULT NULL COMMENT '扩展数据（如图片URL、文件大小等）',
    is_recalled TINYINT NOT NULL DEFAULT 0 COMMENT '是否撤回：0-否，1-已撤回',
    recall_at DATETIME DEFAULT NULL COMMENT '撤回时间',
    is_deleted TINYINT NOT NULL DEFAULT 0 COMMENT '是否删除：0-否，1-删除（只是自己看不到）',

    delivered_at DATETIME DEFAULT NULL COMMENT '送达时间',

    UNIQUE KEY uk_message_key (message_key),
    KEY idx_conversation_messages (conversation_id, is_deleted, sent_at),

-- 消息ID使用雪花算法生成，确保趋势递增

## 设计要点总结

### 1. 会话Key生成策略

```sql
-- 单聊会话Key：确保任意两个用户的会话Key唯一
conversation_key = CONCAT(MIN(user_id_1, user_id_2), '_', MAX(user_id_1, user_id_2))
-- 示例：用户1和用户5的会话Key为 "1_5"

-- 群聊会话Key：直接使用群ID
conversation_key = CONCAT('group_', group_id)
-- 示例：群ID为100的会话Key为 "group_100"
```

### 2. 未读消息数维护

```sql
-- 发送消息时更新会话未读数
UPDATE t_chat_conversation
SET unread_count = unread_count + 1,
    last_message_id = ?,
    last_message_content = ?,
    last_message_time = NOW()
WHERE conversation_key = ?;

-- 标记消息已读时清零未读数
UPDATE t_chat_conversation
SET unread_count = 0
WHERE id = ? AND owner_id = ?;

-- 同时需要更新群组成员的最后已读消息ID
UPDATE t_chat_group_member
SET last_read_message_id = ?, last_read_time = NOW()
WHERE group_id = ? AND user_id = ?;
```

### 3. 消息撤回实现

```sql
-- 撤回消息（2分钟内可撤回）
UPDATE t_chat_message
SET is_recalled = 1,
    recall_at = NOW(),
    content = '该消息已撤回'
WHERE id = ? AND sender_id = ?
AND sent_at >= DATE_SUB(NOW(), INTERVAL 2 MINUTE);

-- 撤回后更新会话的最后一条消息
UPDATE t_chat_conversation c
INNER JOIN (
    SELECT conversation_id, MAX(id) as last_valid_id
    FROM t_chat_message
    WHERE conversation_id = ? AND is_recalled = 0 AND is_deleted = 0
    GROUP BY conversation_id
) r ON c.id = r.conversation_id
SET c.last_message_id = r.last_valid_id;
```

### 4. 离线消息推送

```sql
-- 查询用户离线期间的消息
SELECT * FROM t_chat_message
WHERE receiver_id = ?
AND conversation_type = 1  -- 单聊
AND sent_at > ?  -- 用户最后活跃时间
AND is_recalled = 0
ORDER BY sent_at ASC;

-- 群聊需要查询用户加入群之后的离线消息
SELECT m.* FROM t_chat_message m
INNER JOIN t_chat_group_member gm ON m.receiver_id = gm.group_id
WHERE gm.user_id = ?
AND m.conversation_type = 2  -- 群聊
AND m.sent_at > gm.joined_at
AND m.sent_at > ?  -- 用户最后活跃时间
AND m.sent_at > COALESCE(gm.last_read_time, gm.joined_at)
AND m.sender_id != ?  -- 排除自己发的消息
AND m.is_recalled = 0
ORDER BY m.sent_at ASC;
```

### 5. 历史消息分页查询

```sql
-- 下拉加载更多历史消息
SELECT * FROM t_chat_message
WHERE conversation_id = ?
AND is_deleted = 0
AND sent_at < ?  -- 上一页最后一条消息的时间
ORDER BY sent_at DESC
LIMIT 20;

-- 使用消息ID分页（更准确）
SELECT * FROM t_chat_message
WHERE conversation_id = ?
AND is_deleted = 0
AND id < ?  -- 上一页最后一条消息的ID
ORDER BY id DESC
LIMIT 20;
```

### 6. 群消息广播优化

```sql
-- 发送群消息时的优化策略
-- 1. 消息只写入一条记录
INSERT INTO t_chat_message (...) VALUES (...);

-- 2. 异步更新所有群成员的会话列表
-- 使用消息队列处理，避免同步阻塞

-- 3. 异步更新未读消息数
UPDATE t_chat_conversation
SET unread_count = unread_count + 1,
    last_message_id = ?,
    last_message_content = ?,
    last_message_time = ?
WHERE owner_id IN (
    SELECT user_id FROM t_chat_group_member
    WHERE group_id = ? AND is_quited = 0 AND user_id != ?
);
```
